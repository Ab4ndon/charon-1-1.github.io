#include <stdio.h>
#include <iostream>
using namespace std;

template<class T>
class SortableList
{
    public:
        SortableList(int mSize,int n,T*a);
        ~SortableList();
        int BSearch(const T&x,int left,int right)const;  //把整个函数修饰为const，意思是“函数体内不能对成员数据做任何改动”。
        int BSearch_DieDai(const T&x,int n) const;
       void Merge(int left,int mid,int right);
       void MergeSort();
       void MergeSort(int left,int right);
       void Output();

       void Swap(int &a, int &b);
       int Partition(int left,int right);
       void QuickSort();
       void QuickSort(int left,int right);

    private:
        T*l;
        int maxSize;
        int num;

};
//构造函数
template<class T>
SortableList<T>::SortableList(int mSize,int n,T*a)
{
    l=a;
    maxSize=mSize;
    num=n;
}
//析构函数
template<class T>
SortableList<T>::~SortableList()
{
    delete[]l;
}

/*********************************** 二分搜索 ****************************/
//采用递归的二分搜索
template<class T>
int SortableList<T> ::BSearch(const T&x,int left,int right)const
{
    if(left<=right)
    {
        int m=(left+right)/2;
        if(x<l[m]) 
            return BSearch(x,left,m-1);
        else if(x>l[m])
            return BSearch(x,m+1,right);
        else return m;
    }
    return -1;
}
//采用迭代的二分搜索（尾递归改迭代）
template<class T>
int SortableList<T>::BSearch_DieDai(const T&x,int n)const
{
    int m,left=0,right=n-1;
    while(left<=right)
    {
        m=(left+right)/2;
        if(x<l[m])  
            right=m-1;
        else if(x>l[m])  
            left=m+1;
        else 
            return m;
    }
    return -1;
}

/******************************* 合  并  排  序 ************************************/

template <class T>
void SortableList<T>::MergeSort()
{
    MergeSort(0,num-1);
}
template<class T>
void SortableList<T>::MergeSort(int left,int right)
{
    if(left<right)
    {
        int mid = (left+right)/2;
        MergeSort(left,mid);
        MergeSort(mid+1,right);
        Merge(left,mid,right);
    }
}

template <class T>
void SortableList<T>::Merge(int left, int mid, int right)
{                                      //a数组为待排序的数组 
	T* temp =new T[right - left + 1]; //temp数组用来暂时保存排序好的数组 
	int i = left, j = mid + 1, k = 0; //left 至mid为左半数组 ，mid+1至right为右半数组 
	while((i <= mid)&&(j <= right))  //对应第一步 当左半数组和右半数组都不为空时 
		if(l[i] <= l[j])             //输出两个数组中的较小者 
			temp[k ++] = l[i ++];
		else
			temp[k ++] = l[j ++];
    while(i <= mid)                  //当一数组为空时，输出另一数组的剩余元素 
        temp[k ++] = l[i ++];
    while(j <= right)                //当一数组为空时，输出另一数组的剩余元素 
        temp[k ++] = l[j ++];
    for(i = 0, k = left; k <= right;) //将temp数组赋值给原数组 
        l[k ++] = temp[i ++];
}
 

template<class T>
void SortableList<T>::Output()
{
	for(int i = 0; i < num; i++)
		cout << l[i] << " ";
	cout << endl << endl;
}

/********************************** 快  速  排  序 ***********************************/

template<class T>
void SortableList<T>::Swap(int &a, int &b)
{
	int t = a;
	a = b;
	b = t;
}

template <class T>
int SortableList<T>::Partition(int left,int right)
{
    if(left>=right)
    {
        cout<<"下标范围有误，请确保left<right!"<<endl;
        return 0;
    }
       
    int i = left,j = right +1;
    do 
    {
        do i++; while (l[i]<l[left]);
        do j--; while (l[j] > l[left]);
        if(i < j) Swap(i,j);
    }while(i < j);
    Swap(left,j);
    return j;
}

template <class T>
void SortableList<T>:: QuickSort()
{
    QuickSort(0,num-1);
}

template<class T>
void SortableList<T>:: QuickSort(int left,int right)
{
    if(left<right)
    {
        int j = Partition(left,right);   
        if(j-left<=right-j)                  //为了减少系统栈的深度，一般都会将长的子序列的上下界进栈         
        {                                    //先处理短的子序列
            if(j-left>1)                     
            {
                QuickSort(left,j-1);
                QuickSort(j+1,right);
            }
        }else
        { 
            if(right-j>1)
            {
                QuickSort(j+1,right);
                QuickSort(left,j-1);
            }
        }
    }
}

/************************************ main 函数 ***********************************/
int main()
{
    int a[5]={2,3,1,4,5};
    SortableList<int> list =SortableList<int>(5,5,a);
    int k = list.BSearch(4,0,4);
    int j = list.BSearch_DieDai(4,6);
    cout<<k;
    cout<<j;
     cout << "排序前："<<endl;

    cout << "归并排序后："<<endl;
	// for (int i = 0; i < 5; i++)

	// cout << endl;
    list.QuickSort();
    list.Output();
    cout << "快速排序后："<<endl;
    list.Output();
    return 0;
}
